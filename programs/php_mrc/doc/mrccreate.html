<?php
// --- Create a MRC file
$x=500;
$y=500;
$n=$x*$y;
$data = array();
for ($i=0; $i<$n; $i++){
    $v = ($v > $x) ? 0 : $v;
    $data[$i] = $v;
    $v++;
}

$src_mrc = mrccreate($x,$y);
mrcputdata($src_mrc, $data);
/* update MRC header information based on the inserted data:
    min
    max
    stdev
    avg
    ...etc..
*/
mrcupdateheader($src_mrc);
mrcwrite($src_mrc, "mymrc.mrc"); // save to file "mymrc.mrc"
mrcdestroy($src_mrc);
?>
