<?php
// --- Read a MRC file and display as a PNG
$filename="mymrc.mrc";
$src_mrc = mrcread($filename); // create a mrc resource
$img = mrctoimage($src_mrc); // create a gd image resource from $src_mrc
header("Content-type: image/x-png"); // create png image
imagepng($img);
// -- destroy resources
mrcdestroy($src_mrc);
imagedestroy($img);
?>
