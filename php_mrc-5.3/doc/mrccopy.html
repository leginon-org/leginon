<?php
$x=512;
$y=512;
$mrc_src = mrcread($filename);
mrcbinning($mrc_src,2);
$info = mrcgetinfo($mrc_src);
$src_x = $info[nx];
$src_y = $info[ny];
$mrc_dst = mrccreate($x,$y);
// --- copy part of the image source
mrccopy($mrc_dst, $mrc_src, 200,10,70,10,$src_x/2, $src_y/2);
mrcupdateheader($mrc_dst);
$im = mrctoimage($mrc_dst);
mrcdestroy($mrc_src);
mrcdestroy($mrc_dst);
header("Content-type: image/x-png");
imagepng($im);
imagedestroy($im);
?>
